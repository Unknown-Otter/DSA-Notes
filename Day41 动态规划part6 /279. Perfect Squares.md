class Solution {
    public int numSquares(int n) {

     int max  = Integer.MAX_VALUE; 
     int[] dp = new int[n +1];

     for(int j =0; j < dp.length; j++){
        dp[j] = max; 
     }

     dp[0] = 0;

     for(int i =1; i*i <= n; i++){

        for(int j = i*i; j<=n; j++){
            dp[j]= Math.min(dp[j],dp[j-i*i]+1);
        }
     }

        
   return dp[n];




    }
}
